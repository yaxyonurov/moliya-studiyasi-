CREATE TABLE tokenlar (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  token TEXT UNIQUE NOT NULL,
  ism TEXT,
  muddat DATE,
  izoh TEXT,
  faol BOOLEAN DEFAULT true,
  qurilma TEXT,
  yaratilgan TIMESTAMPTZ DEFAULT now()
);

-- Hamma o'qiy olsin (kirish sahifasi uchun)
ALTER TABLE tokenlar ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Hamma o'qiy oladi" ON tokenlar
  FOR SELECT USING (true);

CREATE POLICY "Hamma yoza oladi" ON tokenlar
  FOR ALL USING (true);
